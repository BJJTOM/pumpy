```typescript
function InfoRow({ 
  label, 
  value, 
  valueColor,  // 값 색상 (상태 색상용)
  bgColor      // 배경색 (체육관 정보용)
}: { 
  label: string; 
  value: string; 
  valueColor?: string; 
  bgColor?: string 
}) {
  return (
    <View style={{ 
      backgroundColor: bgColor || '#f8f9fa' 
    }}>
      <Text>{label}</Text>
      <Text style={{ 
        color: valueColor || '#333' 
      }}>{value}</Text>
    </View>
  )
}
```
