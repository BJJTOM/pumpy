cd /home/ubuntu/pumpy
git pull origin main

# 백엔드 재시작
cd gym_api
source venv/bin/activate
pip install -r requirements.txt
python manage.py migrate
python manage.py collectstatic --noinput
sudo systemctl restart gunicorn

# 프론트엔드 재시작
cd ../gym_web
npm install
npm run build
pm2 restart gym_web
pm2 save
